44 bool Unpacker::BuildRawEvent() {
45  if (!rawEvent.empty())
46  ClearRawEvent();
47 
48  if (numRawEvt == 0) {// This is the first rawEvent. Do some special processing.
49  // Find the first XiaData event. The eventList is time sorted by module.
50  // The first component of each deque will be the earliest time from that module.
51  // The first event time will be the minimum of these first components.
52  if (!GetFirstTime(firstTime))
53  return false;
54  std::cout << "BuildRawEvent: First event time is " << firstTime << " clock ticks.\n";
55  eventStartTime = firstTime;
56  } else {
57  // Move the event window forward to the next valid channel fire.
58  if (!GetFirstTime(eventStartTime))
59  return false;
60  }
61 
62  realStartTime = eventStartTime + eventWidth_;
63  realStopTime = eventStartTime;
64 
65  unsigned int mod, chan;
66  string type, subtype, tag;
67  XiaData *current_event = NULL;
68 
69  // Loop over all time-sorted modules.
70  for (vector<deque<XiaData *> >::iterator iter = eventList.begin(); iter != eventList.end(); iter++) {
71  if (iter->empty())
72  continue;
73 
74  // Loop over the list of channels that fired in this buffer
75  while (!iter->empty()) {
76  current_event = iter->front();
77  mod = current_event->GetModuleNumber();
78  chan = current_event->GetChannelNumber();
79 
80  if (mod > MAX_PIXIE_MOD ||
81  chan > MAX_PIXIE_CHAN) { // Skip this channel
82  cout << "BuildRawEvent: Encountered non-physical Pixie ID (mod = "
83  << mod << ", chan = " << chan << ")\n";
84  delete current_event;
85  iter->pop_front();
86  continue;
87  }
88 
89  double currtime = current_event->GetTime();
90 
91  // Check for backwards time-skip. This is un-handled currently and needs fixed CRT!!!
92  if (currtime < eventStartTime)
93  cout << "BuildRawEvent: Detected backwards time-skip from start=" << eventStartTime << " to "
94  << current_event->GetTime() << "???\n";
95 
96  // If the time difference between the current and previous event is
97  // larger than the event width, finalize the current event, otherwise
98  // treat this as part of the current event
99  if ((currtime - eventStartTime) > eventWidth_)
100  break;
101 
102  // Check for the minimum time in this raw event.
103  if (currtime < realStartTime)
104  realStartTime = currtime;
105 
106  // Check for the maximum time in this raw event.
107  if (currtime > realStopTime)
108  realStopTime = currtime;
109 
110  // Update raw stats output with the new event before adding it to the raw event.
111  RawStats(current_event);
112 
113  // Push this channel event into the rawEvent.
114  rawEvent.push_back(current_event);
115 
116  // Remove this event from the event list but do not delete it yet.
117  // Deleting of the channel events will be handled by clearing the rawEvent.
118  iter->pop_front();
119  }
120  }
121 
122  numRawEvt++;
123 
124  return true;
125 }
