237 pair<double, double> XiaListModeDataDecoder::CalculateTimeInSamples(const XiaListModeDataMask &mask,
238  const XiaData &data) {

241  if (data.GetCfdFractionalTime() == 0 || data.GetCfdForcedTriggerBit())

256  }
