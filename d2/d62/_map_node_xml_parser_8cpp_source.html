142 void MapNodeXmlParser::ParseCalibrations(const pugi::xml_node &node, const ChannelConfiguration &cfg,
143  const bool &isVerbose) {
144  for (pugi::xml_node cal = node; cal; cal = cal.next_sibling(node.name())) {
145  string model = cal.attribute("model").as_string("None");
146  double min = cal.attribute("min").as_double(0);
147  double max = cal.attribute("max").as_double(numeric_limits<double>::max());
148 
149  stringstream pars(cal.text().as_string());
150  vector<double> parameters;
151  while (true) {
152  double p;
153  pars >> p;
154  if (pars)
155  parameters.push_back(p);
156  else
157  break;
158  }
159 
160  if (isVerbose) {
161  sstream_ << node.name() << " Model : " << model << " from [" << min << "," << max << "] ";
162  if (!parameters.empty()) {
163  sstream_ << "with coefficients :";
164  for (vector<double>::iterator itPars = parameters.begin(); itPars != parameters.end(); ++itPars)
165  sstream_ << " " << (*itPars);
166  }
167  messenger_.detail(sstream_.str(), 2);
168  sstream_.str("");
169  }
170 
171  if (strcmp(node.name(), "Calibration") == 0)
172  calibrations_.AddChannel(cfg, model, min, max, parameters);
173  if (strcmp(node.name(), "Walk") == 0)
174  walkCorrector_.AddChannel(cfg, model, min, max, parameters);
175  }
