205  continue;
206 
207  // find the matching low gain event
208  vector<ChanEvent *>::const_iterator itLow = lowEvents.begin();
209  for (; itLow != lowEvents.end(); itLow++) {

212  }
213  }
214  if (itLow != lowEvents.end()) {

216  plot(calib::DD_CLOVER_ENERGY_RATIO, location, ratio * 10.);
217  if ((ratio < lowRatio_ ||

219  continue;
220  }
221  geEvents_.push_back(*itHigh);
222  }
223 
224  // now we sort the germanium events according to their corrected time
225  sort(geEvents_.begin(), geEvents_.end());
