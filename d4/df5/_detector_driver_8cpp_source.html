185 int DetectorDriver::ThreshAndCal(ChanEvent *chan, RawEvent &rawev) {
186  ChannelConfiguration chanCfg = chan->GetChanID();
187  int id = chan->GetID();
188  string type = chanCfg.GetType();
189  string subtype = chanCfg.GetSubtype();
190  set<string> tags = chanCfg.GetTags();
191  bool hasStartTag = chanCfg.HasTag("start");
192  Trace &trace = chan->GetTrace();
193 
194  RandomInterface *randoms = RandomInterface::get();
195 
196  double energy = 0.0;
197 
198  if (type == "ignore" || type == "")
199  return (0);
200 
201  if (!trace.empty()) {
202  plot(D_HAS_TRACE, id);
203 
204  for (vector<TraceAnalyzer *>::iterator it = vecAnalyzer.begin(); it != vecAnalyzer.end(); it++)
205  (*it)->Analyze(trace, chanCfg);
206 
207  //We are going to handle the filtered energies here.
208  vector<double> filteredEnergies = trace.GetFilteredEnergies();
209  if (filteredEnergies.empty()) {
210  energy = chan->GetEnergy() + randoms->Generate();
211  } else {
212  energy = filteredEnergies.front();
213  plot(D_FILTER_ENERGY + id, energy);
214  }
215 
216  //Saves the time in nanoseconds
217  chan->SetHighResTime((trace.GetPhase() * Globals::get()->GetAdcClockInSeconds() +
218  chan->GetTimeSansCfd() * Globals::get()->GetFilterClockInSeconds()) * 1e9);
219  } else {
222  energy = chan->GetEnergy() + randoms->Generate();
223  chan->SetHighResTime(0.0);
224  }
225 
227  double time, walk_correction;
228  if (chan->GetHighResTimeInNs() == 0.0) {
229  time = chan->GetTime(); //time is in clock ticks
230  walk_correction = walk_->GetCorrection(chanCfg, energy);
231  } else {
232  time = chan->GetHighResTimeInNs(); //time here is in ns
233  walk_correction = walk_->GetCorrection(chanCfg, trace.GetQdc());
234  }
235 
236  chan->SetCalibratedEnergy(cali_->GetCalEnergy(chanCfg, energy));
237  chan->SetWalkCorrectedTime(time - walk_correction);
238 
239  rawev.GetSummary(type)->AddEvent(chan);
240  DetectorSummary *summary;
241 
242  summary = rawev.GetSummary(type + ':' + subtype, false);
243  if (summary != NULL)
244  summary->AddEvent(chan);
245 
246  if (hasStartTag && type != "logic") {
247  summary = rawev.GetSummary(type + ':' + subtype + ':' + "start", false);
248  if (summary != NULL)
249  summary->AddEvent(chan);
250  }
251  return (1);
252 }
