111 bool VandleProcessor::Process(RawEvent &event) {

115  plot(D_DEBUGGING, 30);

119  static const vector<ChanEvent *> &betaStarts = event.GetSummary("beta_scint:beta")->GetList();

130  BarBuilder startBars(doubleBetaStarts);
131  startBars.BuildBars();
132  barStarts_ = startBars.GetBarMap();
133 
134  for (BarMap::iterator it = bars_.begin(); it != bars_.end(); it++) {
135  TimingDefs::TimingIdentifier barId = (*it).first;
136  BarDetector bar = (*it).second;
137 
138  if (!bar.GetHasEvent())
139  continue;
140 
141  if (!doubleBetaStarts.empty())
142  AnalyzeBarStarts(bar, barId.first);
143  else
144  AnalyzeStarts(bar, barId.first);
145  }
146
