175 vector<TraceAnalyzer *> DetectorDriverXmlParser::ParseAnalyzers(const pugi::xml_node &node) {
176  std::vector < TraceAnalyzer * > vecAnalyzer;
177 
178  for (pugi::xml_node analyzer = node; analyzer; analyzer = analyzer.next_sibling(node.name())) {
179  string name = analyzer.attribute("name").value();
180  messenger_.detail("Loading " + name);
181 
182  if (name == "CfdAnalyzer") {
183  vecAnalyzer.push_back(new CfdAnalyzer(analyzer.attribute("type").as_string("poly")));
184  } else if (name == "FittingAnalyzer") {
185  vecAnalyzer.push_back(new FittingAnalyzer(analyzer.attribute("type").as_string("gsl")));
186  } else if (name == "TauAnalyzer") {
187  vecAnalyzer.push_back(new TauAnalyzer());
188  } else if (name == "TraceExtractor") {
189  vecAnalyzer.push_back(new TraceExtractor(analyzer.attribute("type").as_string(""),
190  analyzer.attribute("subtype").as_string(""),
191  analyzer.attribute("tag").as_string("trace")));
192  } else if (name == "TraceFilterAnalyzer") {
193  vecAnalyzer.push_back(new TraceFilterAnalyzer(analyzer.attribute("FindPileup").as_bool(false)));
194  } else if (name == "WaaAnalyzer") {
195  vecAnalyzer.push_back(new WaaAnalyzer());
196  } else if (name == "WaveformAnalyzer") {
197  std::vector <std::string> tokens =
198  StringManipulation::TokenizeString(analyzer.attribute("ignored").as_string(""), ",");
199  vecAnalyzer.push_back(new WaveformAnalyzer(std::set<std::string>(tokens.begin(), tokens.end())));
200  } else {
201  stringstream ss;
202  ss << "DetectorDriverXmlParser: Unknown analyzer : " << name;
203  throw PaassException(ss.str());
204  }
205  PrintAttributeMessage(analyzer);
206  }
207  return vecAnalyzer;
