142 void ScopeUnpacker::ProcessRawEvent() {
143  XiaData *current_event = NULL;
144 
145  // Fill the processor event deques with events
146  while (!rawEvent.empty()) {
147  if (!running)
148  break;
149 
150  //Get the first event in the FIFO.
151  current_event = rawEvent.front();
152  rawEvent.pop_front();
153 
154  // Safety catches for null event or empty ->GetTrace().
155  if (!current_event || current_event->GetTrace().empty())
156  continue;
157 
158  if (current_event->GetModuleNumber() != mod_ &&
159  current_event->GetChannelNumber() != chan_)
160  continue;
161 
162  pair<double, double> baseline = CalculateBaseline(current_event->GetTrace(), make_pair(0, 10));
163  pair<double, double> maximum = FindMaximum(current_event->GetTrace(), current_event->GetTrace().size());
164  double qdc = CalculateQdc(current_event->GetTrace(), make_pair(5, 15));
165 
166  if (maximum.second < threshLow_ || (threshHigh_ > threshLow_ && maximum.second > threshHigh_)) {
167  delete current_event;
168  continue;
169  }
170 
171  //Convert the XiaData object into a ProcessedXiaData object
172  ProcessedXiaData *channel_event = new ProcessedXiaData(*current_event);
173 
174  channel_event->GetTrace().SetBaseline(baseline);
175  channel_event->GetTrace().SetMax(maximum);
176  channel_event->GetTrace().SetQdc(qdc);
177 
178  //Push the channel event into the deque.
179  chanEvents_.push_back(channel_event);
180 
181  // Handle the individual XiaData.
182  if (chanEvents_.size() >= numAvgWaveforms_)
183  ProcessEvents();
184  }
185 }
